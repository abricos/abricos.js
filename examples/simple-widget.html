Description

Include abricos.js and helloworld.js on the page if you haven't already. 

<script src="assets/js/abricos.js"></script>

<!-- Abricos.mod.example.HelloWorldWidget -->
<script src="js/example/helloworld.js"></script>

HTML

First we'll create the HTML shell for our widget. 

<div id="helloworld">Loading...</div>

JavaScript

<script>
Abricos.use(function(){
	// Get a container
	var el = document.getElementById('helloworld');
	
	// Get component namespace
	var NS = Abricos.ns('org.abricosjs.example');
	
	// Define in helloworld.js 
	new NS.HelloWorldWidget(el);
});
</script>
