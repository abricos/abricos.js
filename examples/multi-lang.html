This example demonstrates the use of a simple widget HelloWorldWidget
Try clicking on the buttons to change the style of the inscription "Hello World!"

Description

Include abricos.js and helloworld.js on the page if you haven't already. 

<script src="assets/js/abricos.js"></script>

<!-- Abricos.mod.example.HelloWorldWidget -->
<script src="js/example/helloworld.js"></script>

HTML

We'll create the HTML shell for our widgets. 

<table style="width: 100%"><tbody><tr>
<td style=>
	<b>Widget in English</b>
	<div id="helloworld-en">Loading widget. Please, wait...</div>
</td>
<td>
	<b>Виджет на Русском языке</b>
	<div id="helloworld-ru">Загрузка виджета. Пожалуйста, подождите...</div>
</td>
</tr></tbody></table>

JavaScript

<script>
Abricos.use(function(){
	// Get a container for widget in English
	var elEN = document.getElementById('helloworld-en');
	
	// Get component namespace
	var NS = Abricos.ns('org.abricosjs.example');

	// Define in helloworld.js 
	new NS.HelloWorldWidget(elEN, {
		'lang': 'en'
	});
	
	// Get a container for widget in Russian
	var elRU = document.getElementById('helloworld-ru');
	
	// Define in helloworld.js 
	new NS.HelloWorldWidget(elRU, {
		'lang': 'ru'
	});
});
</script>
